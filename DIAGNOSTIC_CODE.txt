// Quick diagnostic check - add to server/src/routes/qbo.ts temporarily

router.get('/debug', requireAuth, async (req: any, res) => {
  try {
    res.json({
      authenticated: true,
      userId: req.userId,
      clerkUserId: req.clerkUserId,
      hasDbConnection: !!sql,
      qbConfigValid: !!(process.env.INTUIT_CLIENT_ID && process.env.INTUIT_CLIENT_SECRET),
      environment: process.env.NODE_ENV,
    });
  } catch (error) {
    res.status(500).json({ 
      error: error instanceof Error ? error.message : 'Unknown error',
      stack: error instanceof Error ? error.stack : undefined
    });
  }
});
